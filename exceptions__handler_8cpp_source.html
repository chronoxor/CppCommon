423  static void GetExceptionPointers(DWORD dwExceptionCode, EXCEPTION_POINTERS** ppExceptionPointers)
424  {
425  CONTEXT ContextRecord;
426  memset(&ContextRecord, 0, sizeof(CONTEXT));
427 
428  EXCEPTION_RECORD ExceptionRecord;
429  memset(&ExceptionRecord, 0, sizeof(EXCEPTION_RECORD));
430 
431 #if defined(_M_IX86)
432  // On x86, we reserve some extra stack which won't be used. That is to
433  // preserve as much of the call frame as possible when the function with
434  // the buffer overrun entered __security_check_cookie with a JMP instead
435  // of a CALL, after the calling frame has been released in the epilogue
436  // of that function.
437  volatile ULONG dw[(sizeof(CONTEXT) + sizeof(EXCEPTION_RECORD)) / sizeof(ULONG)];
438 
439  // Save the state in the context record immediately. Hopefully, since
440  // opts are disabled, this will happen without modifying ECX, which has
441  // the local cookie which failed the check.
442  __asm
443  {
444  mov dword ptr [ContextRecord.Eax ], eax
445  mov dword ptr [ContextRecord.Ecx ], ecx
446  mov dword ptr [ContextRecord.Edx ], edx
447  mov dword ptr [ContextRecord.Ebx ], ebx
448  mov dword ptr [ContextRecord.Esi ], esi
449  mov dword ptr [ContextRecord.Edi ], edi
450  mov word ptr [ContextRecord.SegSs], ss
451  mov word ptr [ContextRecord.SegCs], cs
452  mov word ptr [ContextRecord.SegDs], ds
453  mov word ptr [ContextRecord.SegEs], es
454  mov word ptr [ContextRecord.SegFs], fs
455  mov word ptr [ContextRecord.SegGs], gs
456  pushfd
457  pop [ContextRecord.EFlags]
458 
459  // Set the context EBP/EIP/ESP to the values which would be found
460  // in the caller to __security_check_cookie.
461  mov eax, [ebp]
462  mov dword ptr [ContextRecord.Ebp], eax
463  mov eax, [ebp + 4]
464  mov dword ptr [ContextRecord.Eip], eax
465  lea eax, [ebp + 8]
466  mov dword ptr [ContextRecord.Esp], eax
467 
468  // Make sure the dummy stack space looks referenced.
469  mov eax, dword ptr dw
470  }
471 
472  ContextRecord.ContextFlags = CONTEXT_CONTROL;
473  ExceptionRecord.ExceptionAddress = (PVOID)(ULONG_PTR)ContextRecord.Eip;
474 #elif defined(_M_X64)
475  ULONG64 ControlPc;
476  ULONG64 EstablisherFrame;
477  ULONG64 ImageBase;
478  PRUNTIME_FUNCTION FunctionEntry;
479  PVOID HandlerData;
480 
481  RtlCaptureContext(&ContextRecord);
482 
483  ControlPc = ContextRecord.Rip;
484  FunctionEntry = RtlLookupFunctionEntry(ControlPc, &ImageBase, nullptr);
485 
486  if (FunctionEntry != nullptr)
487  RtlVirtualUnwind(UNW_FLAG_NHANDLER, ImageBase, ControlPc, FunctionEntry, &ContextRecord, &HandlerData, &EstablisherFrame, nullptr);
488 
489 #if defined(__GNUC__)
490  void* return_address = __builtin_return_address(0);
491  ContextRecord.Rip = (ULONGLONG)return_address;
492  ContextRecord.Rsp = (ULONGLONG)__builtin_extract_return_addr(return_address) + 8;
493 #elif defined(_MSC_VER)
494  ContextRecord.Rip = (ULONGLONG)_ReturnAddress();
495  ContextRecord.Rsp = (ULONGLONG)_AddressOfReturnAddress() + 8;
496 #endif
497  ExceptionRecord.ExceptionAddress = (PVOID)ContextRecord.Rip;
498 #else
499  #error Unsupported architecture
500 #endif
501  ExceptionRecord.ExceptionCode = dwExceptionCode;
502 #if defined(__GNUC__)
503  ExceptionRecord.ExceptionAddress = __builtin_return_address(0);
504 #elif defined(_MSC_VER)
505  ExceptionRecord.ExceptionAddress = _ReturnAddress();
506 #endif
507 
508  CONTEXT* pContextRecord = new CONTEXT;
509  memcpy(pContextRecord, &ContextRecord, sizeof(CONTEXT));
510 
511  EXCEPTION_RECORD* pExceptionRecord = new EXCEPTION_RECORD;
512  memcpy(pExceptionRecord, &ExceptionRecord, sizeof(EXCEPTION_RECORD));
513 
514  *ppExceptionPointers = new EXCEPTION_POINTERS;
515  (*ppExceptionPointers)->ContextRecord = pContextRecord;
516  (*ppExceptionPointers)->ExceptionRecord = pExceptionRecord;
517  }
