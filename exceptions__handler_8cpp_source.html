418  static void GetExceptionPointers(DWORD dwExceptionCode, EXCEPTION_POINTERS** ppExceptionPointers)
419  {
420  CONTEXT ContextRecord;
421  memset(&ContextRecord, 0, sizeof(CONTEXT));
422 
423  EXCEPTION_RECORD ExceptionRecord;
424  memset(&ExceptionRecord, 0, sizeof(EXCEPTION_RECORD));
425 
426 #if defined(_M_IX86)
427  // On x86, we reserve some extra stack which won't be used. That is to
428  // preserve as much of the call frame as possible when the function with
429  // the buffer overrun entered __security_check_cookie with a JMP instead
430  // of a CALL, after the calling frame has been released in the epilogue
431  // of that function.
432  volatile ULONG dw[(sizeof(CONTEXT) + sizeof(EXCEPTION_RECORD)) / sizeof(ULONG)];
433 
434  // Save the state in the context record immediately. Hopefully, since
435  // opts are disabled, this will happen without modifying ECX, which has
436  // the local cookie which failed the check.
437  __asm
438  {
439  mov dword ptr [ContextRecord.Eax ], eax
440  mov dword ptr [ContextRecord.Ecx ], ecx
441  mov dword ptr [ContextRecord.Edx ], edx
442  mov dword ptr [ContextRecord.Ebx ], ebx
443  mov dword ptr [ContextRecord.Esi ], esi
444  mov dword ptr [ContextRecord.Edi ], edi
445  mov word ptr [ContextRecord.SegSs], ss
446  mov word ptr [ContextRecord.SegCs], cs
447  mov word ptr [ContextRecord.SegDs], ds
448  mov word ptr [ContextRecord.SegEs], es
449  mov word ptr [ContextRecord.SegFs], fs
450  mov word ptr [ContextRecord.SegGs], gs
451  pushfd
452  pop [ContextRecord.EFlags]
453 
454  // Set the context EBP/EIP/ESP to the values which would be found
455  // in the caller to __security_check_cookie.
456  mov eax, [ebp]
457  mov dword ptr [ContextRecord.Ebp], eax
458  mov eax, [ebp + 4]
459  mov dword ptr [ContextRecord.Eip], eax
460  lea eax, [ebp + 8]
461  mov dword ptr [ContextRecord.Esp], eax
462 
463  // Make sure the dummy stack space looks referenced.
464  mov eax, dword ptr dw
465  }
466 
467  ContextRecord.ContextFlags = CONTEXT_CONTROL;
468  ExceptionRecord.ExceptionAddress = (PVOID)(ULONG_PTR)ContextRecord.Eip;
469 #elif defined(_M_X64)
470  ULONG64 ControlPc;
471  ULONG64 EstablisherFrame;
472  ULONG64 ImageBase;
473  PRUNTIME_FUNCTION FunctionEntry;
474  PVOID HandlerData;
475 
476  RtlCaptureContext(&ContextRecord);
477 
478  ControlPc = ContextRecord.Rip;
479  FunctionEntry = RtlLookupFunctionEntry(ControlPc, &ImageBase, nullptr);
480 
481  if (FunctionEntry != nullptr)
482  RtlVirtualUnwind(UNW_FLAG_NHANDLER, ImageBase, ControlPc, FunctionEntry, &ContextRecord, &HandlerData, &EstablisherFrame, nullptr);
483 
484 #if defined(__GNUC__)
485  void* return_address = __builtin_return_address(0);
486  ContextRecord.Rip = (ULONGLONG)return_address;
487  ContextRecord.Rsp = (ULONGLONG)__builtin_extract_return_addr(return_address) + 8;
488 #elif defined(_MSC_VER)
489  ContextRecord.Rip = (ULONGLONG)_ReturnAddress();
490  ContextRecord.Rsp = (ULONGLONG)_AddressOfReturnAddress() + 8;
491 #endif
492  ExceptionRecord.ExceptionAddress = (PVOID)ContextRecord.Rip;
493 #else
494  #error Unsupported architecture
495 #endif
496  ExceptionRecord.ExceptionCode = dwExceptionCode;
497 #if defined(__GNUC__)
498  ExceptionRecord.ExceptionAddress = __builtin_return_address(0);
499 #elif defined(_MSC_VER)
500  ExceptionRecord.ExceptionAddress = _ReturnAddress();
501 #endif
502 
503  CONTEXT* pContextRecord = new CONTEXT;
504  memcpy(pContextRecord, &ContextRecord, sizeof(CONTEXT));
505 
506  EXCEPTION_RECORD* pExceptionRecord = new EXCEPTION_RECORD;
507  memcpy(pExceptionRecord, &ExceptionRecord, sizeof(EXCEPTION_RECORD));
508 
509  *ppExceptionPointers = new EXCEPTION_POINTERS;
510  (*ppExceptionPointers)->ContextRecord = pContextRecord;
511  (*ppExceptionPointers)->ExceptionRecord = pExceptionRecord;
512  }
