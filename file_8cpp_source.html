679 bool File::IsFileExists() const
680 {
681 #if defined(unix) || defined(__unix) || defined(__unix__) || defined(__APPLE__)
682  struct stat status;
683  int result = stat(string().c_str(), &status);
684  if (result != 0)
685  {
686  if ((errno == ENOENT) || (errno == ENOTDIR))
687  return false;
688  else
689  throwex FileSystemException("Cannot get the status of the file!").Attach(*this);
690  }
691 
692  if (S_ISDIR(status.st_mode))
693  return false;
694  else if (S_ISREG(status.st_mode))
695  return true;
696  else if (S_ISBLK(status.st_mode))
697  return true;
698  else if (S_ISCHR(status.st_mode))
699  return true;
700  else if (S_ISFIFO(status.st_mode))
701  return true;
702  else if (S_ISSOCK(status.st_mode))
703  return true;
704  else
705  return true;
706 #elif defined(_WIN32) || defined(_WIN64)
707  DWORD attributes = GetFileAttributesW(wstring().c_str());
708  if (attributes == INVALID_FILE_ATTRIBUTES)
709  return false;
710 
711  return (attributes & FILE_ATTRIBUTE_DIRECTORY) == 0;
712 #endif
713 }
